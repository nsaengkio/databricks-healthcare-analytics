from pyspark.sql.functions import *

enc = spark.read.format("delta").load("/mnt/healthcare/bronze/encounters")

enc_clean = (
    enc.withColumn("EncounterDate", to_date("EncounterDate"))
       .withColumn("IsNoShow", when(col("Status") == "NoShow", 1).otherwise(0))
       .withColumn(
           "LOS_Category",
           when(col("LengthOfStay") >= 3, "Long Stay")
           .when(col("LengthOfStay") == 0, "Same Day")
           .otherwise("Short Stay")
       )
)

enc_clean.write.format("delta").mode("overwrite").save("/mnt/healthcare/silver/encounters")
